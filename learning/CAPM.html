CAPM Regression 

Ri_train <- train[, colnames(train) != "GSPC"]
Rm_train <- train[, "GSPC"]

capm_results <- list()
summaries <- list()

for (i in colnames(Ri_train)) {
  y <- as.numeric(Ri_train[, i])
  x <- as.numeric(Rm_train)
  model <- lm(y ~ x)
  s <- summary(model)
  summaries[[i]] <- s
  coefs <- coef(s)
  
  capm_results[[i]] <- c(
    alpha = coefs[1, "Estimate"],
    beta = coefs[2, "Estimate"]
  )
}
capm_table <- as.data.frame(do.call(rbind, capm_results))
rownames(capm_table) <- names(capm_results)

CAPM Regression Summary 

pretty_table <- data.frame(
  Asset = character(),
  Alpha = character(),
  Beta = character(),
  t_Alpha = character(),
  t_Beta = character(),
  p_Alpha = character(),
  p_Beta = character(),
  R2 = numeric(),
  stringsAsFactors = FALSE
)

for (i in colnames(Ri_train)) {
  model <- lm(as.numeric(Ri_train[, i]) ~ as.numeric(Rm_train))
  s <- summary(model)
  coefs <- coef(s)

  alpha_str <- sprintf("%.3f<br>(%.3f)", coefs[1, "Estimate"], coefs[1, "Std. Error"])
  beta_str  <- sprintf("%.3f<br>(%.3f)", coefs[2, "Estimate"], coefs[2, "Std. Error"])

  t_Alpha_val <- round(coefs[1, "t value"], 3)
  t_Beta_val  <- round(coefs[2, "t value"], 3)
  p_Alpha_val <- round(coefs[1, "Pr(>|t|)"], 3)
  p_Beta_val  <- round(coefs[2, "Pr(>|t|)"], 3)

  # Bold p-values < 0.05 (statistically significant)
  p_Alpha_fmt <- ifelse(p_Alpha_val < 0.05, sprintf("<b>%.3f</b>", p_Alpha_val), sprintf("%.3f", p_Alpha_val))
  p_Beta_fmt  <- ifelse(p_Beta_val < 0.05, sprintf("<b>%.3f</b>", p_Beta_val), sprintf("%.3f", p_Beta_val))

  pretty_table <- rbind(pretty_table, data.frame(
    Asset = i,
    Alpha = alpha_str,
    Beta = beta_str,
    t_Alpha = sprintf("%.3f", t_Alpha_val),
    t_Beta = sprintf("%.3f", t_Beta_val),
    p_Alpha = p_Alpha_fmt,
    p_Beta = p_Beta_fmt,
    R2 = round(s$r.squared, 3)
  ))
}

colnames(pretty_table) <- c(
  "Asset", 
  "&alpha;", "&beta;", 
  "t(&alpha;)", "t(&beta;)", 
  "p(&alpha;)", "p(&beta;)", 
  "R&sup2;"
)

Security Market Line 

mean_returns <- colMeans(Ri_train)
mean_market <- mean(Rm_train)
r_f <- 0 # Simplifying assumption

betas <- capm_table$beta
names(betas) <- rownames(capm_table)

x_range <- range(betas) * c(0.95, 1.05)
y_range <- range(mean_returns) * c(0.95, 1.05)

plot(betas, mean_returns, pch = 19, col = "blue",
     xlab = "Beta", ylab = "Average Return",
     xlim = x_range, ylim = y_range,
     main = "Security Market Line (SML)")

abline(a = r_f, b = mean_market, col = "red", lwd = 2)
grid(col = "gray")
text(betas, mean_returns, labels = names(betas), pos = 4, offset = 0.3, cex = 0.8)

legend("topleft", legend = c("Actual Returns", "CAPM SML"),
       col = c("blue", "red"), pch = c(19, NA),
       lty = c(NA, 1), lwd = c(NA, 2))

Note that through the SML, we cannot claim that there is mispricing as \(\alpha_i\) is statistically insignificant for all assets. At most, since \(\beta_i\) is statistically significant, we could possibly claim that the SML aligns with the average returns of each asset \(i\).

market_test <- as.numeric(test$GSPC)
actual_returns <- test[, colnames(test) != "GSPC"]

for (asset in colnames(actual_returns)) {
  alpha <- capm_results[[asset]]["alpha"]
  beta <- capm_results[[asset]]["beta"]
  predicted <- alpha + beta * market_test
  actual <- actual_returns[, asset]

  plot(predicted, actual,
       main = paste("Test Sample:", asset),
       xlab = "Predicted Return", ylab = "Actual Return",
